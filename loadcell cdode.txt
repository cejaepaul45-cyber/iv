#include <WiFi.h>
#include <WebServer.h>
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include "HX711.h"

// 1. Local Wi-Fi Configuration
const char* ssid = "ESP32_IV_Scale";
const char* password = "Password123";

// 2. Hardware Pin Definitions for ESP32
const int LOADCELL_DOUT_PIN = 22; // Moved away from Pin 3
const int LOADCELL_SCK_PIN = 23;  // Moved away from Pin 2

// 3. Object Initializations
LiquidCrystal_I2C lcd(0x27, 16, 2); 
HX711 scale;
WebServer server(80);

float calibration_factor = -20.0; // Preserving your exact working factor!

// --- EMBEDDED WEB DASHBOARD ---

// --- WEB ROUTE HANDLERS ---
void handleRoot() {
  server.send(200, "text/html", HTML_DASHBOARD);
}

void handleGetVolume() {
  float avg_ml = scale.get_units(5); 
  if (avg_ml < 0) avg_ml = 0.0;
  server.send(200, "text/plain", String(avg_ml, 0)); // Send zero-decimal number to web
}

void setup() {
  Serial.begin(115200);

  // Initialize the I2C LCD (ESP32 uses D21 for SDA, D22 for SCL automatically)
  Wire.begin(21, 22); 
  lcd.init();                      
  lcd.backlight();
  lcd.setCursor(0, 0);
  lcd.print("Zeroing Scale...");

  // Initialize the Load Cell
  scale.begin(LOADCELL_DOUT_PIN, LOADCELL_SCK_PIN);
  scale.set_scale(calibration_factor);
  scale.tare(); 
  
  lcd.clear();
  Serial.println("Scale zeroed.");

  // Broadcast ESP32 Standalone Wi-Fi Network
  WiFi.softAP(ssid, password);
  IPAddress IP = WiFi.softAPIP();
  Serial.print("Connect to Wi-Fi. Access Dashboard at URL: ");
  Serial.println(IP); 

  // Setup Web Server Routes
  server.on("/", handleRoot);
  server.on("/get_volume", handleGetVolume);
  server.begin();
}

void loop() {
  server.handleClient(); // Handle web dashboard requests

  // Simultaneously update the physical LCD screen layout
  if (scale.is_ready()) {
    float avg_ml = scale.get_units(5);
    if (avg_ml < 0) avg_ml = 0.0;

    lcd.setCursor(0, 0);
    lcd.print("IV Fluid Vol:   ");
    lcd.setCursor(0, 1);
    lcd.print(String(avg_ml, 0) + " ml       "); 
  } else {
    lcd.setCursor(0, 0);
    lcd.print("HX711 Error!    ");
  }

  delay(200); // Quick refresh balance
}
